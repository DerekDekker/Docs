from cryptography.fernet import Fernet
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
import time


# 生成密钥
key = Fernet.generate_key()

# 加载密钥
x = Fernet(key)

# 加密
x.encrypt_at_time(data=b"明文", current_time=int(time.time()))  # current_time 可选

# 解密
x.decrypt(b'秘文')
